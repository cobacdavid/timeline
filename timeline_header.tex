\def\filldebut((#1,#2),#3,#4,#5,#6){
  \fill[#6] (#1,#2) -- ++(#3,0) -- ++(#5,#4/2)
  -- ++(-#5,#4/2)-- ++(-#3,0) -- cycle;
}

\def\fillmilieu((#1,#2),#3,#4,#5,#6){
  \fill[#6] (#1,#2) -- ++(#3,0) -- ++(#5,#4/2) -- ++(-#5,#4/2)
  -- ++(-#3,0) -- ++(#5,-#4/2) --cycle;
}

\def\fillfin((#1,#2),#3,#4,#5,#6){
  \fill[#6](#1,#2) -- ++(#3,0) -- ++(0,#4)
  -- ++(-#3,0) -- ++(#5,-#4/2) -- cycle;
}

\def\filltriBas((#1,#2),#3,#4,#5,#6){
  \fill[#6](#1,#2) -- ++(#3,0) -- ++(-#3/2,#4) --cycle;
}

\def\filltriHaut((#1,#2),#3,#4,#5,#6){
  \fill[#6](#1,#2+#4) -- ++(#3,0) -- ++(-#3/2,-#4) --cycle;
}

\def\drawdebut((#1,#2),#3,#4,#5,#6){
  \draw[#6,line width=5mm] (#1,#2) -- ++(#3,0) -- ++(#5,#4/2)
  -- ++(-#5,#4/2)-- ++(-#3,0) -- cycle;
}

\def\drawmilieu((#1,#2),#3,#4,#5,#6){
  \draw[#6,line width=5mm] (#1,#2) -- ++(#3,0) -- ++(#5,#4/2) -- ++(-#5,#4/2)
  -- ++(-#3,0) -- ++(#5,-#4/2) --cycle;
}

\def\drawfin((#1,#2),#3,#4,#5,#6){
  \draw[#6,line width=5mm](#1,#2) -- ++(#3,0) -- ++(0,#4)
  -- ++(-#3,0) -- ++(#5,-#4/2) -- cycle;
}

\def\drawdebutSansFleche((#1,#2),#3,#4,#5,#6){
  \draw[#6,line width=5mm] (#1+#3, #2) -- (#1,#2)
  -- ++(#5,#4/2) -- ++(-#5,#4/2) -- ++(#3,0);

}

\def\drawmilieuSansFleche((#1,#2),#3,#4,#5,#6){
  \draw[#6,line width=5mm] (#1,#2) -- ++(#3,0);
  \draw[#6,line width=5mm] (#1,#2+#4) -- ++(#3,0);
}

\def\drawfinSansFleche((#1,#2),#3,#4,#5,#6){
  \draw[#6,line width=5mm](#1,#2) -- ++(#3,0)
  -- ++(#5,#4/2) -- ++(-#5,#4/2) -- ++(-#3,0);
}


\def\drawtriBas((#1,#2),#3,#4,#5,#6){

  \draw[#6,line width=2bp](#1,#2+#4) -- ++(#3,0) -- ++(-#3/2,-#4) --cycle;
}
