\def\debut((#1,#2),#3,#4,#5,#6){
  \fill[#6] (#1,#2) -- ++(#3,0) -- ++(#5,#4/2)
  -- ++(-#5,#4/2)-- ++(-#3,0) -- cycle;
}

\def\fin((#1,#2),#3,#4,#5,#6){
  \fill[#6](#1,#2) -- ++(#3,0) -- ++(0,#4)
  -- ++(-#3,0) -- ++(#5,-#4/2) -- cycle;
}

\def\milieu((#1,#2),#3,#4,#5,#6){
  \fill[#6] (#1,#2) -- ++(#3,0) -- ++(#5,#4/2) -- ++(-#5,#4/2)
  -- ++(-#3,0) -- ++(#5,-#4/2) --cycle;
}
