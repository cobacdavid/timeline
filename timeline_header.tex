\def\debut((#1,#2),#3,#4,#5,#6){
  \fill[#6] (#1,#2) -- ++(#3,0) -- ++(#5,#4/2)
  -- ++(-#5,#4/2)-- ++(-#3,0) -- cycle;
}

\def\milieu((#1,#2),#3,#4,#5,#6){
  \fill[#6] (#1,#2) -- ++(#3,0) -- ++(#5,#4/2) -- ++(-#5,#4/2)
  -- ++(-#3,0) -- ++(#5,-#4/2) --cycle;
}

\def\fin((#1,#2),#3,#4,#5,#6){
  \fill[#6](#1,#2) -- ++(#3,0) -- ++(0,#4)
  -- ++(-#3,0) -- ++(#5,-#4/2) -- cycle;
}

\def\triBas((#1,#2),#3,#4,#5,#6){
  \fill[#6](#1,#2) -- ++(#3,0) -- ++(-#3/2,#4) --cycle;
}

\def\triHaut((#1,#2),#3,#4,#5,#6){
  \fill[#6](#1,#2+#4) -- ++(#3,0) -- ++(-#3/2,-#4) --cycle;
}

% fonctions de debuggage
\def\trait((#1,#2),(#3,#4){
  \draw[fill=red,<->] (#1,#2) -- (#3,#4);
